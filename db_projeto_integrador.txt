create database projeto_integrador;

create table usuario(
Email text not null,
Senha int not null,
ID_usuario int primary key,
Nome text not null
);
 create table questionario(
 ID_questionario int primary key,
 Titulo text not null,
 Descricao text not null
 );
create table respostas(
 ID_resposta int primary key,
 ID_questionario int,
 constraint ID_questionario foreign key (ID_questionario)
 references questionario(ID_questionario),
 respostas text not null,
 Pergunta_01_Vinculo_Senac text not null,
 Pergunta_02_Idade text not null,
 Pergunta_03_Renda_Mensal text not null,
 Pergunta_04_Gastos_Mensais text not null,
 Pergunta_05_salario_Investido text not null,
 Pergunta_06_Percade_Investimento text not null,
 Pergunta_07_Uso_do_dinheiro text not null,
 Pergunta_08_Patrimonio_Investido text not null,
 Pergunta_09_Opcoes text not null,
 Pergunta_10_Gastos_Por_Impulso text not null,
 Pergunta_11_Decisoes_financeiras text not null,
 Pergunta_12_Tempo_Investido text not null,
 Pergunta_13_Reserva_Emergencial text not null,
 Pergunta_14_Mercado_Financeiro text not null,
 Pergunta_15_Produtos_Financeiros text not null
 );
create table perfil(
ID_perfil int primary key,
ID_usuario int,
constraint ID_usuario foreign key (ID_usuario)
references usuario (ID_usuario),
Nome text not null,
Perfil text not null,
Descricao text not null,
vinculo text not null,
Perfil_IA text,
Analise_IA text
);

create table resultado(
ID_resultado int primary key,
ID_perfil int,
constraint ID_perfil_resultado foreign key(ID_perfil)
references perfil (ID_perfil),
ID_usuario int,
constraint ID_usuario_resultado foreign key(ID_usuario)
references usuario (ID_usuario),
ID_questionario int not null,
constraint ID_questionario_resultado foreign key(ID_questionario)
references questionario (ID_questionario)
);
